# SinkSonic — DietPi automation settings
# Copy this file and Automation_Custom_Script.sh to the SD card
# boot partition BEFORE first boot:
#
#   cp dietpi.txt               /media/boot/dietpi.txt
#   cp Automation_Custom_Script.sh /media/boot/Automation_Custom_Script.sh
#
# DietPi reads dietpi.txt on first boot and applies everything
# automatically. No SSH or manual steps required.

# ── SinkSonic-specific overrides ──────────────────────────────────────────────
# Hostname the device advertises on the network
AUTO_SETUP_NET_HOSTNAME=SinkSonic

# SSH server: -1=Dropbear | -2=OpenSSH
AUTO_SETUP_SSH_SERVER_INDEX=-1

# Install Docker (DietPi software ID 134)
AUTO_SETUP_INSTALL_SOFTWARE_ID=134

# Install PipeWire audio stack via APT
AUTO_SETUP_APT_INSTALLS=pipewire pipewire-pulse wireplumber pulseaudio-utils avahi-daemon docker-compose-v2

# Post-install script on boot partition (handles PipeWire config,
# Docker data migration, SinkSonic container, overlayfs)
AUTO_SETUP_CUSTOM_SCRIPT_EXEC=0

# Fully automated — no prompts during first boot
AUTO_SETUP_AUTOMATED=1

# Enable built-in analog audio on Raspberry Pi (3.5mm jack)
# For HDMI: vc4-hdmi | For USB: USB-Audio
CONFIG_SOUNDCARD=snd_bcm2835
